   497→export function getProductsByCategorySlug(categorySlug: string): Product[] {
   498→  const cat = categories.find((c) => c.slug === categorySlug);
   499→  if (!cat) return [];
   500→  // If it has children, include products from all children
   501→  if (cat.children?.length) {
   502→    const childIds = cat.children.map((c) => c.id);
   503→    return products.filter((p) => p.categoryId && childIds.includes(p.categoryId));
   504→  }
   505→  // Otherwise, return products directly in this category
   506→  const parentCat = categories.find((c) => c.children?.some(ch => ch.slug === categorySlug));
   507→  if (parentCat) {
   508→    const child = parentCat.children!.find((c) => c.slug === categorySlug);
   509→    if (child) return products.filter((p) => p.categoryId === child.id);
   510→  }
   511→  return products.filter((p) => p.categoryId === cat.id);
   512→}
