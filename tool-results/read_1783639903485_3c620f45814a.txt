  1311→export function getProductsByCategorySlug(categorySlug: string): Product[] {
  1312→  const cat = categories.find((c) => c.slug === categorySlug);
  1313→  if (!cat) return [];
  1314→  // If it has children, include products from all children
  1315→  if (cat.children?.length) {
  1316→    const childIds = cat.children.map((c) => c.id);
  1317→    return products.filter((p) => p.categoryId && childIds.includes(p.categoryId));
  1318→  }
  1319→  // Otherwise, return products directly in this category
  1320→  const parentCat = categories.find((c) => c.children?.some(ch => ch.slug === categorySlug));
  1321→  if (parentCat) {
  1322→    const child = parentCat.children!.find((c) => c.slug === categorySlug);
  1323→    if (child) return products.filter((p) => p.categoryId === child.id);
  1324→  }
  1325→  return products.filter((p) => p.categoryId === cat.id);
  1326→}
